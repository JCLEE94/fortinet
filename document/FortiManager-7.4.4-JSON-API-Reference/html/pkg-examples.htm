Policy Package Examples


Policy package management


Add multiple folders and policy packages under the root ADOM.

Request: 

{
    "method": "set",
    "params": [ 
        {
            "url": "pm/pkg/adom/root",
            "data": [ 
                {
                    "name": "folder01",
                    "type": "folder",
                    "subobj": [ 
                        {
                            "name": "subfolder01",
                            "type": "folder" 
                        },
                        {
                            "name": "subfolder02",
                            "type": "folder",
                            "subobj": [ 
                                {
                                    "name": "ssfolder01",
                                    "type": "folder",
                                    
                                },
                                {
                                    "name": "pkg01",
                                    "type": "pkg",
                                    "scope member": [ 
                                        {
                                            "name": "All_FortiGate" 
                                        },
                                        {
                                            "name": "fg1",
                                            "vdom": "root" 
                                        },
                                        {
                                            "name": "fg2",
                                            "vdom": "vd1" 
                                        } 
                                    ] 
                                } 
                            ] 
                        } 
                    ] 
                },
                {
                    "name": "folder02",
                    "type": "folder" 
                } 
            ] 
        } 
    ],
    "id": 1
}

Response: 

{
    "result": [ 
        {
            "url": "pm/pkg/adom/root",
            "status": {
                "code": 0,
                "message": "OK" 
            } 
        } 
    ],
    "id": 1
}

Query all package objects inside a folder including scope member.

Request: 

{
    "method": "get",
    "params": [  
        {
            "url": "pm/pkg/adom/root/folder01/subfolder02",
            "fields": [ "scope member" ]  
        }  
    ],
    "id": 2
}

Response: 

{
    "result": [ 
        {
            "url": "pm/pkg/adom/root/folder01/subfolder02",
            "data": [ 
                {
                    "name": "ssfolder01",
                    "type": "folder" 
                },
                {
                    "name": "pkg1",
                    "obj ver": 16,
                    "oid": 1030,
                    "package settings": {
                        "central-nat": 0,
                        "fwpolicy-implicit-log": 0,
                        "fwpolicy6-implicit-log": 0,
                        "ngfw-mode": 0 
                    },
                    "scope member": [ 
                        {
                            "name": "All_FortiGate" 
                        },
                        {
                            "name": "fg1",
                            "vdom": "root" 
                        },
                        {
                            "name": "fg2",
                            "vdom": "vd1" 
                        } 
                    ],
                    "type": "pkg" 
                } 
            ],
            "status": {
                "code": 0,
                "message": "OK" 
            } 
        } 
    ],
    "id": 2
}

Request: 

{
    "method": "update",
    "params": [  
        {
            "url": "pm/pkg/adom/root/folder02",
            "data": {
                "name": "new_folder" 
            } 
        }  
    ],
    "id": 3
}

Response: 

{
    "result": [ 
        {
            "url": "pm/pkg/adom/root/folder02",
            "status": {
                "code": 0,
                "message": "OK" 
            } 
        } 
    ],
    "id": 3
}

Rename and update scope for a policy package.

Request: 

{
    "method": "set",
    "params": [  
        {
            "url": "pm/pkg/adom/root/folder01/subfolder02/pkg01",
            "data": {
                "name": "new_pkg",
                "scope member": [ 
                    {
                        "name": "All_FortiGate"  
                    } 
                ] 
            } 
        }  
    ],
    "id": 4
}

Response: 

{
    "result": [ 
        {
            "url": "pm/pkg/adom/root/folder01/subfolder02/pkg01",
            "status": {
                "code": 0,
                "message": "OK" 
            } 
        } 
    ],
    "id": 4
}

                        "name": "All_FortiGate" 
                    }
